doBufView:	jsr	GotoFirstMenu
	jsr	StopTerm
	jsr	OpenRR
	beq	10$
	ldw	A2,#BufSavNo
	jsr	CNCLBox
	jmp	MnuActDn	; we're done.
10$:	jsr	ResColorScn	; <-- only if xfer not done
	jsr	CrsrTL	; Screen clear and top-left
	lda	txtTop
	jsr	ClrTrmScn	; lda from prev MATTERS
20$:	jsr	GetRByte
	cpx	#0
	bne	90$
	jsr	SafeDraw
	bra	20$
90$:	mvb	txtXY+2,R1H
	mvw	txtXY,R11
	jsr	DrwCRLF
	mvb	R1H,txtXY+2
	mvw	R11,txtXY
	jmp	keySetup

BufSavNo:	.byte	27,"Buffer Empty?!",0


doBufDump:	jsr	GotoFirstMenu
	jsr	StopTerm
	jsr	OpenRR
	beq	10$
	ldw	A2,#BufSavNo
	jsr	CNCLBox
	jmp	MnuActDn	; we're done.
10$:	jsr	ResColorScn	; <-- only if xfer not done
	jsr	CrsrTL	; Screen clear and top-left
	lda	txtTop
	jsr	ClrTrmScn	; lda from prev MATTERS
	lda	BufferFlg
	pha
	ldb	BufferFlg,#0
20$:	jsr	GetRByte
	cpx	#0
	bne	90$
	sta	keyData
	jsr	KeyPRtn
	bra	20$
90$:	pla
	sta	BufferFlg
	mvb	txtXY+2,R1H
	mvw	txtXY,R11
	jsr	DrwCRLF
	mvb	R1H,txtXY+2
	mvw	R11,txtXY
	jmp	keySetup
